 11###########################################################
 12# How to create your own model
 13# --------------------------------
 14# Dataset
 15# +++++++++++++++
 16# First, you need a dataset to create your model and a **models** folder to store it.
 17
 18import os
 19
 20dirs = [
 21    "model",
 22    "model/dataset",
 23    "model/dataset/train",
 24    "model/dataset/train/train_x",
 25    "model/dataset/train/train_y",
 26    "model/dataset/test",
 27    "model/dataset/test/test_x",
 28    "model/dataset/test/test_y",
 29    "model/models",
 30]
 31for d in dirs:
 32    os.makedirs(d, exist_ok=True)
 33
 34# This dataset is essential to create your model, you need to add the following :
 35# * ``train_x`` : a training set of images that you hand-cut yourself using `Fiji <https://imagej.net/software/fiji/downloads>`_ and more specifically `ImageJ <https://imagej.net/software/imagej>`_.
 36# * ``train_y`` : the masks for each segmented frame in a zip file containing all the ROI files
 37# * ``test_x`` and ``test_y`` : you should do the same, it will serve to test the model that has been trained on the training set.
 38
 39# Here is an example of an image that will be placed in the **x** sets (wether it is the training or the test set) on the left. And on the right, each yellow lining represent a ROI (region of interest). Those are saved in a zip file and in the **y** sets.
 40# .. raw:: html
 41#
 42#    <p align="center">
 43#      <img src="./doc/souce/images/example_frame.jpg" alt="frame_ex" width="45%"/>
 44#      <img src="./doc/source/images/example_frame_ROI.jpg" alt="ROI_ex" width="45%"/>
 45#    </p>
 46#
 47# .. note::
 48# Documentation on Fiji can be found `here <https://imagej.net/ij/docs/index.html>`_.
 49#
 50# Next, the `kartezio` package that will help us build the model need two other files : `META.json` and `dataset.csv`.
 51#
 52# The META file is here to help the package recognize the formats of the different objects in the dataset folder.
 53
 54import json
 55
 56# Path to the JSON file
 57meta_path = "model/dataset/META.json"
 58
 59# Content of the JSON file
 60meta_data = {
 61    "name": "Macrophage",
 62    "scale": 1.0,
 63    "label_name": "macrophage",
 64    "mode": "dataframe",
 65    "input": {"type": "image", "format": "hsv"},
 66    "label": {"type": "roi", "format": "polygon"},
 67}
 68
 69# Create the JSON file with the content in it
 70with open(meta_path, "w", encoding="utf-8") as f:
 71    json.dump(meta_data, f, indent=4)
 72
 73print(f"Fichier {meta_path} créé avec succès !")
 74
 75##################################################
 76# The csv file is here to help kartezio read your dataset.
 77
 78from Set_up.dataset_csv import create_dataset_csv
 79
 80create_dataset_csv(input_folder="model/dataset", output_csv="model/dataset/dataset.csv")
 81
 82########################################################
 83# Your two files are now created and located in the `model/dataset` folder. We can now train our model.
 84#
 85#
 86# Training
 87# +++++++++++++++
 88# Now that we have the structure kartezio needs to function correctly, we can create and train a model using the `create_segmentation_model` function of the kartezio package. You can find the following code in the `Set_up/train_model.py` file.
 89
 90from kartezio.apps.segmentation import create_segmentation_model
 91from kartezio.endpoint import EndpointThreshold
 92from kartezio.dataset import read_dataset
 93from kartezio.training import train_model
 94
 95import time
 96from datetime import timedelta
 97
 98t0_train = time.time()
 99
100DATASET = "model/dataset"
101OUTPUT = "model/models"
102
103generations = 1000
104_lambda = 5
105frequency = 5
106rate = 0.1
107print(rate)
108model = create_segmentation_model(
109    generations,
110    _lambda,
111    inputs=3,
112    nodes=30,
113    node_mutation_rate=rate,
114    output_mutation_rate=rate,
115    outputs=1,
116    fitness="IOU",
117    endpoint=EndpointThreshold(threshold=4),
118)
119
120dataset = read_dataset(DATASET)
121elite, a = train_model(model, dataset, OUTPUT, callback_frequency=frequency)
122
123t1_train = time.time()
124elapsed_train = int(t1_train - t0_train)
125#########################################
126# Testing
127# +++++++++++++++
128# Now that you trained your model, you can test it to see if it gives good prediction.
129
130import numpy as np
131import pandas as pd
132
133from kartezio.easy import print_stats
134from kartezio.dataset import read_dataset
135from kartezio.fitness import FitnessIOU
136from kartezio.inference import ModelPool
137
138t0_test = time.time()
139
140scores_all = {}
141pool = ModelPool(f"model/models", FitnessIOU(), regex="*/elite.json").to_ensemble()
142dataset = read_dataset(f"model/dataset", counting=True)
143annotations_test = 0
144annotations_training = 0
145roi_pixel_areas = []
146for y_true in dataset.train_y:
147    n_annotations = y_true[1]
148    annotations_training += n_annotations
149for y_true in dataset.test_y:
150    annotations = y_true[0]
151    n_annotations = y_true[1]
152    annotations_test += n_annotations
153    for i in range(1, n_annotations + 1):
154        roi_pixel_areas.append(np.count_nonzero(annotations[annotations == i]))
155print(f"Total annotations for training set: {annotations_training}")
156print(f"Total annotations for test set: {annotations_test}")
157print(f"Mean pixel area for test set: {np.mean(roi_pixel_areas)}")
158
159scores_test = []
160scores_training = []
161for i, model in enumerate(pool.models):
162    # Test set
163    _, fitness, _ = model.eval(dataset, subset="test")
164    scores_test.append(1.0 - fitness)
165
166    # Training set
167    _, fitness, _ = model.eval(dataset, subset="train")
168    scores_training.append(1.0 - fitness)
169
170# the only downside is that the loop computes scores of models you've already tested
171# and you don't need to compute them again, but it's to have a csv file later that summarizes
172# all your models
173
174t1_test = time.time()
175elapsed_test = int(t1_test - t0_test)
176elapsed_total = elapsed_train + elapsed_test
177
178print(scores_training)
179print(scores_test)
180
181scores_all[f"training"] = scores_training
182scores_all[f"test"] = scores_test
183
184pd.DataFrame(scores_all).to_csv("model/scores.csv", index=False)
185
186print(f"Total runtime : {timedelta(seconds=elapsed_total)}")
187print(f"Training time : {timedelta(seconds=elapsed_train)}")
188print(f"Testing time : {timedelta(seconds=elapsed_test)}")
