gettingstarted

 1#%%
 2from locate.locate import locate 
 3from locate.list_sep import segmentation
 4from locate.defuse import defuse, invdefuse
 5from track.track import track
 6from video.inputconfig import inputconfig
 7from video.result import video,result, videocomp
 8import os
 9from analyse.intensity import intensity,intensitymed
10from analyse.distance import distance
11from analyse.size import size
12from analyse.perimeter import perimeter
13from analyse.recap import aggregate
14import shutil
15from track.filtre import supprimer_petit
16
17#%%
18input_folder = "../input/6hpa_fish4" # Enter your data folder name instead of yourdatafolder (../ for interactive window in vscode)
19n = 130 # Number of frame in the initial video 
20p = 10 # minimal number of frame where you can track your macrophage, if it is present in less or equal p frame it will not be tracked
21
22# create each frame of the video into picture folder (red picture are stored in "input/yourdatafolder/dataset/test/test_x", green picture are stored in "input/yourdatafolder/vert/frames")
23frame = inputconfig(input_folder)
24
25#%%
26# Delete the list_sep and list_comp folder if they already exist (to not have a differnet size in the folder than the one you mentionned 'n')
27# as the output folder is common to all the different inputs you can add
28if os.path.exists("output/list_sep"):
29    shutil.rmtree("output/list_sep")
30if os.path.exists("output/list_comp"):
31    shutil.rmtree("output/list_comp")
32#%%
33# Function which will do the segmentation, it will create a folder output with in it "list_comp" which contain the segmentation at each frame and "list_sep" which contain every object in separate picture file at each frame 
34locate(input_folder)
35
36#%%
37# This will stock all the picture in class in order to accelerate the program
38image_storage = segmentation("output/list_sep")
39image_storage.load_images()
40
41#%%
42# This will prevent and separate the merging macrophage 
43image_storage = defuse(n, image_storage)
44image_storage = invdefuse(n, image_storage)
45
46#%%
47# This will do the tracking of every macrophage and filter if you detect a macrophage on enough frame
48track(n, 0.5, image_storage)
49supprimer_petit(p)
50
51#%%
52# This will create in the output folder two video which show the results of the tracking
53result(input_folder)
54video()
55
56#%%
57# This is for deleting non necessary folder to liberate some place 
58shutil.rmtree("output/list_def")
59shutil.rmtree("output/list_sep")
60shutil.rmtree("output/result")
61shutil.rmtree("output/resultv")
62
63#%%
64# This is to create folder who will contain the data 
65if not os.path.exists("output/data"):
66    os.makedirs("output/data")
67if not os.path.exists("output/plot"):
68    os.makedirs("output/plot")
69
70#%%
71# These will collect the data and stock them in dataframe 
72intmed = intensitymed(n,frame, input_folder) # Data on the intensity of macrophage 
73dis = distance(n) # data of distance to the right border
74siz = size(n) # data of the size of the macrophage 
75per = perimeter(n) # data of the perimeter of the macrophage 
76recap = aggregate(dis,intmed,siz,per) # summary of each data for each macrophage in a movie
